```php
$atomic->route('GET /@lang/about', 'App\Http\Controllers\Page->about');

class Page extends Controller {
  public function about($atomic, array $params) {
      $lang = $params['lang'] ?? get_locale();
     
  }
}
```

```html
    <?php foreach (get_languages() as $lang): ?>
      <a class="w3-button w3-small w3-white w3-margin-right"
         href="<?= htmlspecialchars(lang_url('/', $lang), ENT_QUOTES) ?>"
         hreflang="<?= htmlspecialchars($lang, ENT_QUOTES) ?>"
         <?= $lang===get_locale() ? 'aria-current="true"' : '' ?>>
        <?= strtoupper($lang) ?>
      </a>
    <?php endforeach; ?>
```
